273  {
274  // 要素の最大成分を求める
275  float element[4]; // 0:X, 1:Y, 2:Z, 3:W
276  element[0] = m._m[0]._x - m._m[1]._y - m._m[2]._z + 1.0f;
277  element[1] = -m._m[0]._x + m._m[1]._y - m._m[2]._z + 1.0f;
278  element[2] = -m._m[0]._x - m._m[1]._y + m._m[2]._z + 1.0f;
279  element[3] = m._m[0]._x + m._m[1]._y + m._m[2]._z + 1.0f;
280 
281  unsigned int index = 0;
282  for( signed int i=1; i<4; ++i ) {
283  if( element[i] <= element[index] ) continue;
284  index = i;
285  }
286 
287  if( element[index] < 0.0f ) {
288  return false; // 変換失敗
289  }
290 
291  // 最大要素の値を求める
292  float v = sqrtf( element[index] ) * 0.5f;
293  float scale = 0.25f / v;
294 
295  switch ( index ) {
296  case 0: //---- X
297  _x = v;
298  _y = (m._m[0]._y + m._m[1]._x) * scale;
299  _z = (m._m[2]._x + m._m[0]._z) * scale;
300  _w = (m._m[1]._z - m._m[2]._y) * scale;
301  break;
302  case 1: //---- Y
303  _x = (m._m[0]._y + m._m[1]._x) * scale;
304  _y = v;
305  _z = (m._m[1]._z + m._m[2]._y) * scale;
306  _w = (m._m[2]._x - m._m[0]._z) * scale;
307  break;
308  case 2: //---- Z

310  _y = (m._m[1]._z + m._m[2]._y) * scale;
